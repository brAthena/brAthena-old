xmas_in,100,96,4	script	Papai Noel::Santa2	718,{
	mes "[Papai Noel]";
	if (xmas_npc==0) set xmas_npc, 1;
	if (#event_xmas > 0 && #event_xmas < 30 ) goto L_Start;
	mes "Feliz Natal!";
	if (!Class || #event_xmas >= 30) close;
	mes "Eu tenho um presente para você! HoHoHo!";
	getitem rand(664,667),1; //
	set #event_xmas,#event_xmas+1;
	close;

L_Start:
	mes "Estou tendo um pequeno problema...";
	mes "Você se importa em ouvir?";
	next;
	menu "Ouvir o Papai Noel.",M_0, "Dar a prova ao Papai Noel.",M_1, "Cancelar.",M_End;

	M_0:
		mes "[Papai Noel]";
		mes "Meu problema é...";
		mes "Parece haver um homem lá fora, que está me imitando e causando terror.";
		next;
		mes "[Papai Noel]";
		mes "Como o Grinch da lenda, ele está tomando os brinquedos das crianças e guardando-os para si mesmo.";
		next;
		mes "[Papai Noel]";
		mes "Estou muito ocupado aqui embrulhando e organizando meus presentes para o próximo ano, então por isso eu não posso sair e encontrá-lo.";
		mes "Gostaria que você saisse e destruisse esse homem para mim.";
		next;
		mes "[Papai Noel]";
		mes "Ele está com meus sacos mágicos, por isso ele entrará para dentro do saco sempre que você o atacar e escapará para um outro lugar.";
		next;
		mes "[Papai Noel]";
		mes "Porém,";
		mes "Com pressa, ele poderá derrubar algumas coisas.";
		mes "Se ele derrubar uma de suas meias que usa para roubar as crianças, aproveite e pegue-a.";
		next;
		mes "[Papai Noel]";
		mes "Se você coletar 3 dessas meias, irei te premiar";
		mes "com um protótipo de caixa misteriosa";
		mes "que tenho guardado próximo ao laboratório. Ela gera presentes aleatórios e me economiza bastante trabalho.";
		close;
	M_1:
		mes "[Papai Noel]";
		if (countitem(7034) < 3) goto L_NotEnuf;
		delitem 7034,3;
		mes "Parece que você está fazendo um ótimo trabalho";
		mes "derrotando esses Papais Noeis falsos. Continue assim!";
		next;
		getitem 644,1; // Caixa de Presente
		set #event_xmas,#event_xmas+1;
		mes "[Papai Noel]";
		mes "Aqui está seu presente.";
		mes "Se você conseguir mais 3, irei te dar outra Caixa.";
		mes "Espero que continue coletando mais Meias.";
		close;
	
		L_NotEnuf:
			mes "Você não possui meias suficientes como prova.";
			mes "Derrote mais desses Papais Noeis";
			mes "e consiga mais Meias para obter mais prêmios.";
			close;
	M_End:
		mes "[Papai Noel]";
		mes "Entendo. Bem, pelo menos";
		mes "nós nos encontraremos novamente na manhã de Natal.";
		close;

OnInit:
	disablenpc "Papai Noel";
	end;
}
